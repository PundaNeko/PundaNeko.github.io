async def scrape_title(session, link):
    async with session.get("https://runescape.wiki" + link) as response:
        html = await response.text()
        soup = BeautifulSoup(html, "html.parser")
        title = soup.find("span", class_="mw-page-title-main")
        return link, title.get_text(strip=True) if title else "No title"

async def main(links):
    async with aiohttp.ClientSession() as session:
        tasks = [scrape_title(session, link) for link in links]
        results = await asyncio.gather(*tasks, return_exceptions=True)

Calibrator.cs — マウスキャリブレーション（抜粋）
// TGS・展示会では様々なマウスやDPI環境でプレイされるため、
// プレイ開始前にスクリーン座標とワールド座標のズレを計測・補正する仕組みを実装。
void Calibrate()
{
    // Step 1: 画面中央のターゲットでオフセット（絶対ズレ）を計測
    if (!centerCalibrated)
    {
        Vector2 targetLocal, mouseLocal;
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            aimer.canvas.transform as RectTransform,
            Camera.main.WorldToScreenPoint(initialTargetCoords), null, out targetLocal);
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            aimer.canvas.transform as RectTransform,
            Input.mousePosition, null, out mouseLocal);

        // マウス位置とターゲット位置の差分をオフセットとして保存
        aimer.offsetX = targetLocal.x - mouseLocal.x;
        aimer.offsetY = targetLocal.y - mouseLocal.y;
        centerCalibrated = true;
        return;
    }

    // Step 2: 上下左右2点のターゲットでスケール差（感度倍率）を計算
    // マウスの移動量と実際のターゲット間距離の比率を平均して倍率を導出
    float avgYMult = ((mouseY[1] - mouseY[0]) / (targetY[1] - targetY[0])
                    + (mouseY[2] - mouseY[0]) / (targetY[2] - targetY[0])) / 2f;
    float avgXMult = ((mouseX[1] - mouseX[0]) / (targetX[1] - targetX[0])
                    + (mouseX[2] - mouseX[0]) / (targetX[2] - targetX[0])) / 2f;

    // 倍率が1未満（縮小方向）になるケースは想定外なのでクランプ
    aimer.multiplierX = Mathf.Max(avgXMult, 1f);
    aimer.multiplierY = Mathf.Max(avgYMult, 1f);
    aimer.crosshair.gameObject.SetActive(true);
}
GroupObserver.cs — ウェーブ進行管理（抜粋）
// 敵が全滅したときに呼ばれる。次のウェーブ・移動・ダイアログを順番に処理する。
public void ReduceEnemies()
{
    remainingEnemies--;
    if (remainingEnemies > 0) return; // まだ敵が残っている

    outliner.ResetOutline();

    // キャリーオーバータイマーが動いていれば止めて攻撃トラッカーを再開
    if (carryOverCoroutine != null)
    {
        StopCoroutine(carryOverCoroutine);
        carryOverCoroutine = null;
        EnemyAttackTracker.Instance.enabled = true;
    }
    ProcessNextWave();
}

// 次のウェーブタイプ（移動 or スポーン）に応じて処理を分岐するコルーチン
private IEnumerator ProcessNextWaveRoutine(NextGroupType.GroupType nextType)
{
    // グループの最終ウェーブだった場合はスコアを表示してから進む
    if (isNextGroupEnd)
        yield return StartCoroutine(UIController.Instance.levelScore.ShowScore(false));

    if (nextType == NextGroupType.GroupType.Move)
    {
        // プレイヤーを次のスプライン区間へ移動（オンレールの核心部分）
        playerRef.GetComponent<SplineMovement>().ChangeCurrentSpline();
    }
    else if (nextType == NextGroupType.GroupType.Spawn)
    {
        // キューから次のウェーブを取り出してスポーン
        activeWave = enemyWaves[0];
        enemyWaves.RemoveAt(0);
        SpawnEnemies();
    }
}
PointerLocator.cs — 射撃ヒット検出とキャッシュ（抜粋）
// スコープ有無でレイの起点が変わるが、ヒット処理は共通。
// 通常時: スクリーン上の照準位置からレイを飛ばす
// スコープ時: セカンドカメラの向きから直接レイを飛ばす
void HitDetection()
{
    if (!scopeController.scoped) NonScopedHitDetection();
    else                         ScopedHitDetection();
}

void NonScopedHitDetection()
{
    Ray ray = Camera.main.ScreenPointToRay(crosshair.transform.position);
    if (!Physics.Raycast(ray, out RaycastHit hit, Mathf.Infinity, shootableMasks)) return;

    EnemyHitManager enemy = GetCachedEnemy(hit.collider.gameObject);
    if (enemy != null)
    {
        enemy.Shot();
        // ヒットマーカーのフェードが途中なら中断して再生
        if (hitMarkerFade != null) StopCoroutine(hitMarkerFade);
        hitMarkerFade = StartCoroutine(ShowHitMarker());
        if (!enemy.isTarget) Instantiate(bloodEffect, hit.point, Quaternion.LookRotation(hit.normal)).Play();
    }
    Instantiate(sparkEffect, hit.point, Quaternion.LookRotation(hit.normal)).Play();
}

// GetComponent は毎フレーム呼ぶとコストが高いためDictionaryでキャッシュ
private Dictionary<GameObject, EnemyHitManager> enemyCache
    = new Dictionary<GameObject, EnemyHitManager>();

private EnemyHitManager GetCachedEnemy(GameObject go)
{
    if (!enemyCache.TryGetValue(go, out var enemy))
    {
        enemy = go.GetComponentInParent<EnemyHitManager>();
        enemyCache[go] = enemy; // 初回のみ検索、以降はキャッシュから返す
    }
    return enemy;
}
LifeController.cs — ライフ管理プロパティ（抜粋）
// livesをプロパティにすることで、値の変更が必ずこのセッターを通るように強制。
// 上限クランプ・ゲームオーバー判定・HUD再描画を一箇所に集約している。
public float lives
{
    get { return _lives; }
    set
    {
        // 上限を超えた回復は切り捨て
        if (value > maxLives) value = maxLives;

        // 0以下でゲームオーバー画面へ遷移

            UIController.Instance.timerController.ShowResult();
            value = 0;
        }

        RedrawLives(); // 常にHUDと同期
    }
}
